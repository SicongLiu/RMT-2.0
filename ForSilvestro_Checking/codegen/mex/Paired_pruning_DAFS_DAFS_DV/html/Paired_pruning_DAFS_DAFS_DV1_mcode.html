1function [remainQOctave,Dist] = Paired_pruning_DAFS_DAFS_DV(feature1,depdScale1,matches,combineScore,feature2,depdScale2,doctave,toctave )
  2% pruning considering that two feature overlapping in time should have also
  3% same variate
  4    remainQOctave=zeros(doctave,toctave);
  5    Dist=0;
  6    for ii = 1 : doctave
  7        for  jj = 1 : toctave
  8            index1 = find(feature1(5,:) == ii & feature1(6,:) == jj);
  9            matches11 = matches(:, index1);
 10            combineScore11 = combineScore(index1);
 11
 12            [diagY,I] = sort(combineScore11,2,'descend');
 13            diagY_1=diagY;
 14            diagMatch = matches11(:,I);
 15            diagRemainMatch = diagMatch;
 16            counter=0;
 17%             Counteroccurrences= 0
 18       %     i=1;
 19            dimention=size(diagRemainMatch,2);
 20            i=1;
 21            while i<=dimention
 22                centerc = feature1(2,diagRemainMatch(1,i));
 23                centerm = feature2(2,diagRemainMatch(2,i));
 24                rangec = 3*(feature1(4,diagRemainMatch(1,i))) ;
 25                rangem = 3*(feature2(4,diagRemainMatch(2,i))) ;
 26                startc = centerc - rangec ;
 27                endc = centerc + rangec ;
 28                startm = centerm - rangem ;
 29                endm = centerm + rangem ;
 30
 31                keept = true;
 32                keepd = true;
 33
 34                depdin1 = depdScale1(:,diagRemainMatch(1,i));
 35                depdin1 = depdin1(depdin1~=0);
 36                depdin2 = depdScale2(:,diagRemainMatch(2,i));
 37                depdin2 = depdin2(depdin2~=0);
 38                over = overlap (depdin1,depdin2);
 39                if(over < 0.5)
 40                    keepd = false;
 41                end
 42
 43                j = 1;
 44                 
 45              
 46                while  (j<=counter) && (keept == true)&& (keepd == true)%(j<=size(diagRemainMatch,2)) && (keept == true)&& (keepd == true)
 47                    % first step check only time
 48                   
 49                        
 50                        rem_centerc = feature1(2,diagRemainMatch(1,j));
 51                        rem_centerm = feature2(2,diagRemainMatch(2,j));
 52                        rem_rangec = 3*(feature1(4,diagRemainMatch(1,j))) ;
 53                        rem_rangem = 3*(feature2(4,diagRemainMatch(2,j))) ;
 54                        rem_startc = rem_centerc - rem_rangec ;
 55                        rem_endc = rem_centerc + rem_rangec ;
 56                        rem_startm = rem_centerm - rem_rangem ;
 57                        rem_endm = rem_centerm + rem_rangem ;
 58                        
 59                        VariateF1 = feature1(1,diagRemainMatch(1,j));
 60                        VariateF2 = feature2(1,diagRemainMatch(2,j));
 61                        list1 = zeros(4,1);%[];
 62                        list2 = zeros(4,1);%[];
 63
 64
 65                        list1(4-3,1) =rem_startc;% [list1;rem_startc;rem_endc];
 66                        list1(4-2,1) = rem_endc;
 67                        list2(4-3,1) = rem_startm;%[list2;rem_startm;rem_endm];
 68                        list2(4-2,1) = rem_endm;
 69                        list1(4-1,1) = startc;% [list1;startc;endc];
 70                        list1(4,1)   = endc;%
 71                        list2 (4-1,1) = startm;%[list2;startm;endm];
 72                        list2 (4,1) = endm;%
 73
 74                        [c d] = sort(list1);
 75                        rankcstart = find(c==startc);
 76                        rankcend = find(c==endc);
 77                        [c d] = sort(list2);
 78                        rankmstart = find(c==startm);
 79                        rankmend = find(c==endm);
 80
 81                        if (startc == rem_startc) && (endc==rem_endc) && (startm == rem_startm) && (endm == rem_endm)                            
 82                            if(VariateF1 == VariateF2)
 83                                keept = false;                            
 84                                break;
 85                            else
 86                                keept = true;
 87                            end
 88                        else
 89                            if rankcstart(1) == rankmstart(1) && rankcend(1) ==rankmend(1)
 90                                keept = true;
 91                            else
 92                                keept = false;
 93                                break;
 94                            end
 95                        end
 96               
 97                    j = j+1;
 98                end
 99                if (keept == true) && (keepd == true)
100%                     diagRemainMatch(i) =1;% [diagRemainMatch diagMatch(:,i)];
101                    Dist = Dist + diagY_1(i);
102                    counter=counter+1;
103                else
104                    diagRemainMatch(:,i)=[];
105                    diagY_1(:,i)=[];
106                    i=i-1;
107                    dimention=size(diagRemainMatch,2);
108                end
109                i=i+1;
110            end
111            % diagRemainFeature = feature1(:,diagRemainMatch(1,:));
112            TT = counter;% size(diagRemainMatch, 2);
113            remainQOctave(ii,jj) = TT;%[remainQOctave TT];
114        end
115    end
